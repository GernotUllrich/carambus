**Create test/support/t06_tournament_test_helper.rb:**

The T06 helper needs to create a tournament plan that has BOTH group stages AND finals. Since TournamentPlan.default_plan only creates round-robin, we must construct executor_params manually for a T06 plan.

A T06 plan for 6 players (2 groups of 3, then semifinals + final) looks like:
```json
{
  "g1": { "pl": 3, "rs": "eae_ma", "sq": { "r1": { "t1": "1-2" }, "r2": { "t1": "1-3" }, "r3": { "t1": "2-3" } } },
  "g2": { "pl": 3, "rs": "eae_ma", "sq": { "r1": { "t2": "1-2" }, "r2": { "t2": "1-3" }, "r3": { "t2": "2-3" } } },
  "hf1": { "r4": { "t-rand*": ["g1.rk1", "g2.rk2"] } },
  "hf2": { "r4": { "t-rand*": ["g2.rk1", "g1.rk2"] } },
  "fin": { "r5": { "t-rand*": ["hf1.rk1", "hf2.rk1"] } },
  "GK": 9,
  "RK": ["fin.rk1", "fin.rk2", ["hf1.rk2", "hf2.rk2"]]
}
```

```ruby
# frozen_string_literal: true

module T06TournamentTestHelper
  TEST_ID_BASE = 50_000_000

  @@t06_test_counter = 0

  def create_t06_tournament_with_seedings(tournament_attrs = {})
    discipline = disciplines(:carom_3band)
    season = seasons(:current)
    region = regions(:nbv)

    @@t06_test_counter += 1
    tournament_id = TEST_ID_BASE + 30_000 + (@@t06_test_counter * 200)

    # Create a T06 plan (6 players, 2 groups of 3, then semis + final)
    plan = TournamentPlan.find_or_initialize_by(name: "T06_Test_6")
    executor_params = {
      "g1" => { "pl" => 3, "rs" => "eae_ma", "sq" => {
        "r1" => { "t1" => "1-2" },
        "r2" => { "t1" => "1-3" },
        "r3" => { "t1" => "2-3" }
      }},
      "g2" => { "pl" => 3, "rs" => "eae_ma", "sq" => {
        "r1" => { "t2" => "1-2" },
        "r2" => { "t2" => "1-3" },
        "r3" => { "t2" => "2-3" }
      }},
      "hf1" => { "r4" => { "t-rand*" => ["g1.rk1", "g2.rk2"] }},
      "hf2" => { "r4" => { "t-rand*" => ["g2.rk1", "g1.rk2"] }},
      "fin" => { "r5" => { "t-rand*" => ["hf1.rk1", "hf2.rk1"] }},
      "GK" => 9,
      "RK" => ["fin.rk1", "fin.rk2", ["hf1.rk2", "hf2.rk2"]]
    }
    plan.assign_attributes(
      players: 6,
      executor_class: " ",
      executor_params: executor_params.to_json,
      ngroups: 2,
      nrepeats: 1,
      tables: 2
    )
    plan.save!

    tournament = Tournament.create!({
      id: tournament_id,
      title: "Test T06 Tournament",
      season: season,
      organizer: region,
      organizer_type: "Region",
      discipline: discipline,
      state: "initialized",
      date: 2.weeks.from_now,
      balls_goal: 30,
      innings_goal: 25,
      tournament_plan: plan
    }.merge(tournament_attrs))

    players = (1..6).map do |i|
      Player.create!(
        id: tournament_id + i,
        firstname: "T06Test",
        lastname: "Player#{i}",
        ba_id: tournament_id + 1_000_000 + i
      )
    end

    seedings = players.each_with_index.map do |player, idx|
      Seeding.create!(
        id: tournament_id + 1000 + idx,
        tournament: tournament,
        player: player,
        position: idx + 1,
        region_id: region.id
      )
    end

    { tournament: tournament, players: players, seedings: seedings, plan: plan }
  end

  def cleanup_t06_tournament(data)
    return unless data.is_a?(Hash)
    tournament = data[:tournament]
    return unless tournament
    tournament = Tournament.find_by(id: tournament.id)
    return unless tournament

    tournament.games.each { |g| g.game_participations.destroy_all }
    tournament.games.destroy_all
    tournament.seedings.destroy_all
    tournament.reload.tournament_monitor&.destroy
    tournament.destroy
    data[:players]&.each { |p| p.destroy if Player.exists?(p.id) }
    data[:plan]&.destroy if data[:plan]&.persisted?
  end
end
```

**2. Game Creation for T06 (CHAR-03):**
- `test "T06 creates correct group games"` — 6 players, 2 groups of 3: 3 games per group = 6 group games. Assert tournament.games.where("gname like 'group%'").count == 6
- `test "T06 group games have correct gname format"` — assert gnames include "group1:1-2", "group1:1-3", "group1:2-3", "group2:1-2", "group2:1-3", "group2:2-3"
- `test "T06 creates finals games after group phase"` — after group phase, when populate_tables processes endgame keys, games "hf1", "hf2", "fin" should be created. (Note: these are created during populate_tables for the endgame portion, which runs during do_reset_tournament_monitor for KO/endgame plans.) Verify these endgame games exist in tournament.games.

**3. Result Pipeline (CHAR-02):**

Testing the full result pipeline requires simulating a completed game with TableMonitor data. This is complex because report_result expects a TableMonitor in final state with ba_results data. The approach is to test the components individually:

- `test "accumulate_results builds correct rankings from game participations"` — after setup creates group games, manually create GameParticipation records with result data (points, result, innings, gd, hs), call @tm.accumulate_results, assert @tm.data["rankings"]["groups"]["group1"] contains correct player entries with accumulated points/gd
- `test "write_game_result_data writes table_monitor data to game"` — create a mock-like TableMonitor scenario: create a Game, set up a TableMonitor associated to the game with data including "ba_results", call @tm.write_game_result_data(table_monitor), assert game.reload.data contains "ba_results" and "finalized_at"
- `test "update_game_participations_for_game writes correct results to participation"` — create a game with two GameParticipations, prepare table_monitor_data hash with playera/playerb results (result, innings, hs, balls_goal), call @tm.update_game_participations_for_game(game, table_monitor_data), assert gp1.reload.result == expected, gp1.points == expected (winner gets 2, loser gets 0)

**4. Group Phase Detection (CHAR-03):**
- `test "group_phase_finished? returns false when group games have no ended_at"` — assert_not @tm.group_phase_finished?
- `test "group_phase_finished? returns true when all group games ended"` — update all group games with ended_at, assert @tm.group_phase_finished?
