```ruby
require "application_system_test_case"

class TableMonitorBroadcastSmokeTest < ApplicationSystemTestCase
  setup do
    # Build complete fixture chain for get_options! and scoreboard rendering.
    # TableMonitor -> Table -> Location must all be linked.
    # A Game must be assigned for the show action to render (not redirect).
    #
    # Use existing fixtures where possible; create additional records
    # only for associations that fixtures don't provide.
    @table_monitor = table_monitors(:one)
    @table = tables(:one)

    # Ensure table_monitor is in "new" state for the ready! transition
    @table_monitor.update_columns(state: "new", data: "{}")

    # Create a game and assign it so the show action renders the scoreboard
    # (show redirects to locations_path if game_id is blank)
    @game = Game.create!(id: 50_000_100)
    @table_monitor.update_columns(game_id: @game.id)

    # Create minimal GameParticipation records if needed by the view
    # (check what _show.html.erb requires)
  end

  test "AASM state change broadcasts visible DOM update to subscribed browser session" do
    # Visit scoreboard page — this establishes WebSocket subscription to table-monitor-stream
    visit table_monitor_url(@table_monitor, locale: :de)

    # Verify the scoreboard page loaded (not redirected)
    assert_selector "#full_screen_table_monitor_#{@table_monitor.id}"

    # Trigger AASM state change: new -> ready (per D-06)
    @table_monitor.reload
    @table_monitor.ready!

    # Force job execution — test queue adapter does not auto-execute perform_later.
    # The default branch in TableMonitorJob renders the full scoreboard partial
    # and broadcasts inner_html to #full_screen_table_monitor_{id}.
    TableMonitorJob.perform_now(@table_monitor.id)

    # Capybara waits up to default_max_wait_time (10s) for DOM update (per D-05).
    # After the ready! transition, the scoreboard should reflect the "ready" state.
    # Verify the broadcast reached the browser by checking for updated content.
    # The exact assertion depends on what the scoreboard view renders for state "ready".
    # state_display(:de) for "ready" = "Frei"
    assert_selector "#full_screen_table_monitor_#{@table_monitor.id}", text: /Frei/i
  end
end
```
