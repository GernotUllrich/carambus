    Step B — Create `app/javascript/controllers/bk2_kombi_shot_controller.js`:

    ```js
    import { Controller } from "@hotwired/stimulus"

    // BK2-Kombi shot input controller.
    //   - Packs all form fields into the submit button's dataset before the reflex fires
    //   - Generates a UUID shot_sequence_number per click (AdvanceMatchState idempotency guard)
    //   - 500ms debounce on submit (T-38.1-13 double-tap protection)
    //   - B2 FIX: fullPinImage target drives the D-15 `middle_pin_only_from_full_image=2`
    //     scoring path (unreachable without this control)
    export default class extends Controller {
      static targets = ["fallenPins", "middlePinOnly", "fullPinImage",
                        "trueCarom", "falseCarom",
                        "passages", "foul", "foulCode", "bandHit", "submit"]

      connect() { this.lastSubmitAt = 0 }

      submit(event) {
        const now = Date.now()
        if (now - this.lastSubmitAt < 500) { event.preventDefault(); return }
        this.lastSubmitAt = now

        const btn = this.submitTarget
        btn.dataset.fallen_pins = this.fallenPinsTarget.value
        btn.dataset.middle_pin_only = this.middlePinOnlyTarget.checked
        // B2 FIX — populate full_pin_image from the new fullPinImage checkbox target.
        // Without this assignment the reflex reads "false" always and the D-15
        // 2-point rule is unreachable.
        btn.dataset.full_pin_image = this.hasFullPinImageTarget ? this.fullPinImageTarget.checked : false
        btn.dataset.true_carom = this.trueCaromTarget.checked
        btn.dataset.false_carom = this.falseCaromTarget.checked
        btn.dataset.passages = this.passagesTarget.value
        btn.dataset.foul = this.foulTarget.checked
        btn.dataset.foul_code = this.foulTarget.checked ? this.foulCodeTarget.value : ""
        btn.dataset.band_hit = this.hasBandHitTarget ? this.bandHitTarget.checked : false
        btn.dataset.shot_sequence_number = this.generateSequenceNumber()
      }

      generateSequenceNumber() {
    ```
