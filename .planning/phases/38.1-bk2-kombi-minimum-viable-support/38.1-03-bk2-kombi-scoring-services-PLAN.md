```ruby
{
  observations: {
    fallen_pins: 0, # Integer 0..5
    middle_pin_only: false, # Boolean; true iff only the middle pin fell
    true_carom: false, # Boolean
    false_carom: false, # Boolean
    passages: 0, # Integer, typically 0..1 but not hard-capped

  }
}
```

### ScoreShot return shape — D-13 + BK2-KOMBI-RULES.md lines 138-160 literal:

Non-foul:
```ruby
{
  scoring: {
    foul: false,
    points_for_current_player: Integer, # = raw_total
    points_for_opponent: 0,
    raw_total: Integer,
    credited_total: Integer # = raw_total
  },
  transitions: {
    bonus_shot_awarded: Boolean, # true iff Direkter Zweikampf AND non-foul AND (fallen_pins==5 OR fallen_pins==5 && carom)
    turn_ends: Boolean,
    next_player_at_table: "playera" | "playerb",
    next_shots_left_in_turn: Integer | nil # Integer for Direkter Zweikampf, nil for Serienspiel
  }
}
```

Foul:
```ruby
{
  scoring: {
    foul: true,
    foul_code: Symbol,
    foul_points: Integer, # NEGATIVE (the published value, e.g. -6 for wrong_ball). In Perplexity transcript this was stored positive and added; here we store signed to preserve D-16 literal values.
    points_for_current_player: 0,
    points_for_opponent: Integer, # = |raw_total| + |foul_points| — assigned to opponent as a POSITIVE credit; OR the current player receives `raw_total + foul_points` as a NEGATIVE delta, depending on Carambus convention
    raw_total: Integer,
    credited_total: Integer
  },
  transitions: { ... }
}
```

**Carambus convention decision (planner's discretion):** Store foul credit as `points_for_opponent` = positive number going TO opponent (simpler for CableReady broadcast + UI display). AdvanceMatchState then writes +opponent_credit to opponent's set_score. Document this choice in the class-level docstring.

### Bonus-shot rule — BK2-KOMBI-RULES.md lines 181-222 literal:
- Applies ONLY in `current_phase == "direkter_zweikampf"`
- Requires `!scoring[:foul] && (fallen_pins == 5 || (fallen_pins == 5 && (true_carom || false_carom)))`
- Effect: `turn_ends=false, next_player_at_table=current, next_shots_left_in_turn=max(shots_left_in_turn, 1)`

### Serienspiel turn-end rule:
- `turn_ends = scoring[:foul] || scoring[:points_for_current_player].zero?`
- `next_shots_left_in_turn = nil` (not applicable in Serienspiel)

### Direkter Zweikampf non-bonus turn-end rule:
- `next_shots = shots_left_in_turn - 1`
- `turn_ends = scoring[:foul] || next_shots <= 0`
- On turn_ends: player changes, next_shots_left_in_turn resets to 2

### Set close — BK2-KOMBI-RULES.md lines 226-241:
- After applying scoring, check if either playera or playerb in current set has `score >= set_target_points`
- If yes: set is closed. `sets_won[winner] += 1`. Increment `current_set_number`. Reset `current_phase` to `"direkter_zweikampf"`, `shots_left_in_turn` to 2. Kickoff-alternation for the next set is left as planner's discretion (simplest: the player with FEWER sets_won starts, or alternate — use alternate for consistency with the existing Carambus `current_kickoff_player` pattern).

### Match close:
- Match is over when either `sets_won[playera] >= 2` OR `sets_won[playerb] >= 2`
- AdvanceMatchState sets `tm.data['bk2_state']['match_finished'] = true` and `tm.data['bk2_state']['match_winner'] = "playera" | "playerb"`
- Should NOT trigger the existing karambol end-of-game flow (Game.finish!) — that hook is for Plan 04 / post-38.1 polish. For 38.1 dry-run, leaving the flag on bk2_state is sufficient; volunteer scorer announces match end manually.

    # frozen_string_literal: true

    module Bk2Kombi
      # Pure shot evaluator for BK2-Kombi discipline.
      # Takes a shot_payload hash (shape: see phase 38.1 CONTEXT.md D-13)
      # and a state hash (shape: see D-14) and returns { scoring:, transitions: }.
      # No database writes. No mutation of input hashes.
      #
      # Foul convention: points_for_opponent holds the POSITIVE credit awarded to
      # the opposing player; foul_points holds the SIGNED published value per D-16
      # (e.g., -6 for wrong_ball). Rationale: downstream UI + CableReady broadcast
      # find it clearer to render "opponent +6" than "current -6 then credit moves".
      class ScoreShot
        DEFAULT_RULES = {
          scoring: {
            pin: 1,
            middle_pin_only_from_full_image: 2,
            true_carom: 1,
            false_carom_with_pin_or_passage: 1,
            passage: 1
          },
          foul_values: {
            wrong_ball: -6,
            no_object_ball_hit_with_band: -1,
            no_object_ball_hit_no_band: -2,
            premature_shot: -1,
            direct_play_on_pins: nil, # computed: -[1, fallen_pins].max
            other: -1
          }
        }.freeze

        def self.call(shot_payload:, state:, rules: DEFAULT_RULES)
          new(shot_payload: shot_payload, state: state, rules: rules).call
        end

        def initialize(shot_payload:, state:, rules:)
          @shot_payload = shot_payload.deep_symbolize_keys
          @state = state.deep_symbolize_keys
          @rules = rules.deep_symbolize_keys
        end

        def call
          raw = calculate_raw_points
          foul = calculate_foul_points
          scoring = foul[:foul] ? apply_foul_scoring(raw, foul) : apply_positive_scoring(raw)
          transitions = calculate_transitions(scoring)
          { scoring: scoring, transitions: transitions }
        end

        private

        # Implement calculate_raw_points, calculate_foul_points, apply_positive_scoring,
        # apply_foul_scoring, calculate_transitions — following BK2-KOMBI-RULES.md §Rohpunkte,
        # §Fehler, §Punkte zuweisen, §Bonusstoß exactly, but renaming player_a → playera to
        # match Carambus convention.
      end
    end
    ```

    # frozen_string_literal: true

    module Bk2Kombi
      # Applies a single BK2-Kombi shot to a TableMonitor's data['bk2_state'].
      # Delegates pure evaluation to Bk2Kombi::ScoreShot and persists the result.
      # Closes the current set when a player reaches set_target_points.
      # Closes the match when a player reaches 2 sets won (best-of-3).
      class AdvanceMatchState
        DEFAULT_SET_TARGET = 50

        def self.call(table_monitor:, shot_payload:)
          new(table_monitor: table_monitor, shot_payload: shot_payload).call
        end

        def initialize(table_monitor:, shot_payload:)
          @tm = table_monitor
          @shot_payload = shot_payload
        end

        def call
          init_state_if_missing!
          return idempotency_noop if already_applied?

          state = @tm.data["bk2_state"].deep_dup

          result = Bk2Kombi::ScoreShot.call(
            shot_payload: @shot_payload,
            state: state
          )

          apply_scoring!(state, result[:scoring])
          apply_transitions!(state, result[:transitions])
          close_set_if_reached!(state)
          close_match_if_reached!(state)
          record_sequence!(state) if @shot_payload[:shot_sequence_number]

          @tm.data["bk2_state"] = state
          @tm.save!

          { scoring: result[:scoring], transitions: result[:transitions], state: state }
        end

        private

        def init_state_if_missing!
          return if @tm.data["bk2_state"].is_a?(Hash)

          @tm.data["bk2_state"] = {
            "current_set_number" => 1,
            "current_phase" => "direkter_zweikampf",
            "player_at_table" => @tm.data["current_kickoff_player"].presence || "playera",
            "shots_left_in_turn" => 2,
            "set_scores" => {
              "1" => { "playera" => 0, "playerb" => 0 },
              "2" => { "playera" => 0, "playerb" => 0 },
              "3" => { "playera" => 0, "playerb" => 0 }
            },
            "sets_won" => { "playera" => 0, "playerb" => 0 },
            "set_target_points" => derive_set_target
          }
        end

        def derive_set_target
          # Read from game options if present; else DEFAULT_SET_TARGET.
          # Plan 04 / game-setup options can populate this; for 38.1 dry-run default is 50.
          (@tm.data.dig("bk2_options", "set_target_points") || DEFAULT_SET_TARGET).to_i
        end

        # ... apply_scoring!, apply_transitions!, close_set_if_reached!, close_match_if_reached!,
        # already_applied?, record_sequence!, idempotency_noop ...
      end
    end
    ```

    end

    def apply_transitions!(state, transitions)
      state["player_at_table"] = transitions[:next_player_at_table].to_s
      state["shots_left_in_turn"] = transitions[:next_shots_left_in_turn]
    end

    def close_set_if_reached!(state)
      set_no = state["current_set_number"].to_s
      target = state["set_target_points"].to_i
      a = state["set_scores"][set_no]["playera"]
      b = state["set_scores"][set_no]["playerb"]
      return unless a >= target || b >= target

      state["sets_won"][winner] += 1

      state["set_winner_#{set_no}"] = winner
