Step 3 — Extend `Bk2::CommitInning` with the BK-family dispatcher. Refactor the existing `call` method:

```ruby
# app/services/bk2/commit_inning.rb (after rename)
module Bk2
  class CommitInning < ApplicationService
    def initialize(table_monitor:, player:, inning_total:, shot_sequence_number: nil)
      @table_monitor = table_monitor
      @player = player.to_s
      @inning_total = inning_total
      @shot_sequence_number = shot_sequence_number
    end

    def call
      validate!
      state = ensure_state_initialized!
      form = free_game_form(state)
      result = apply_rule_for_form!(form, state)
      persist_and_advance!(state, result[:transitions])
      { state: state, transitions: result[:transitions] }
    end

    private

    def validate!
      raise ArgumentError, "player must be 'playera' or 'playerb'" unless %w[playera playerb].include?(@player)
      raise ArgumentError, "inning_total must be an Integer" unless @inning_total.is_a?(Integer)
      raise ArgumentError, "inning_total out of range (-999..+999)" unless (-999..999).include?(@inning_total)
    end

    def ensure_state_initialized!
      Bk2::AdvanceMatchState.new(@table_monitor).initialize_bk2_state!
      @table_monitor.data["bk2_state"] || {}
    end

    def free_game_form(state)
      discipline = @table_monitor.discipline
      form = if discipline&.data.present?
               (JSON.parse(discipline.data) rescue {})["free_game_form"]
             end
      form ||= @table_monitor.data.is_a?(Hash) ? @table_monitor.data["free_game_form"] : nil
      form
    end

    def apply_rule_for_form!(form, state)
      case form
      when "bk_2plus"
        apply_opponent_credit_rule(state)
      when "bk_2", "bk50", "bk100"
        apply_additive_rule(state)
      when "bk2_kombi"
        phase = state["current_phase"]
        case phase
        when "direkter_zweikampf" then apply_opponent_credit_rule(state)
        when "serienspiel"        then apply_additive_rule(state)
        else raise ArgumentError, "Unknown bk2_kombi current_phase: #{phase.inspect}"
        end
      else
        raise ArgumentError, "Unknown free_game_form: #{form.inspect} for Bk2::CommitInning"
      end
    end

    def apply_opponent_credit_rule(state)
      # BK-2plus + DZ phase of BK-2kombi: negative total credits abs value to opponent.
      current_set = state["current_set_number"].to_s
      opponent = @player == "playera" ? "playerb" : "playera"
      if @inning_total >= 0
        state["set_scores"][current_set][@player] = (state["set_scores"][current_set][@player].to_i + @inning_total)
      else
        state["set_scores"][current_set][opponent] = (state["set_scores"][current_set][opponent].to_i + @inning_total.abs)
      end
      { transitions: [] }
    end

    def apply_additive_rule(state)
      # BK-2 / BK50 / BK100 / SP phase of BK-2kombi: signed add to current player's set score.
      current_set = state["current_set_number"].to_s
      state["set_scores"][current_set][@player] = (state["set_scores"][current_set][@player].to_i + @inning_total)
      { transitions: [] }
    end

    def persist_and_advance!(state, transitions)
      @table_monitor.data["bk2_state"] = state
      @table_monitor.save!
      close_result = Bk2::AdvanceMatchState.new(@table_monitor).close_set_if_reached!
      transitions.concat(Array(close_result[:transitions])) if close_result.is_a?(Hash)
    end
  end
```
